    const GAS_CODE = `function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var currentQuota = MailApp.getRemainingDailyQuota();
    
    if (data.action === "ping") {
      return ContentService.createTextOutput(JSON.stringify({status: "ok", quota: currentQuota})).setMimeType(ContentService.MimeType.JSON);

      var aliases = GmailApp.getAliases();

      if (mainEmail && aliases.indexOf(mainEmail) === -1) {
        aliases.unshift(mainEmail);
      }
      aliases = aliases.filter(function(a) { return a && a.length > 0; });
      return ContentService.createTextOutput(JSON.stringify({aliases: aliases, quota: currentQuota})).setMimeType(ContentService.MimeType.JSON);
    }

    if (data.action === "saveData") {
      var sheet = getOrCreateDatabase();
      var payloadStr = JSON.stringify(data.payload);
      sheet.clearContents();
      
      // Chunk data to bypass Google Sheets 50,000 character limit per cell
      var chunks = [];
      var chunkSize = 40000;
      for (var i = 0; i < payloadStr.length; i += chunkSize) {
         chunks.push([payloadStr.substring(i, i + chunkSize)]);
      }
      sheet.getRange(1, 1, chunks.length, 1).setValues(chunks);
      
      return ContentService.createTextOutput(JSON.stringify({status: "saved", quota: currentQuota})).setMimeType(ContentService.MimeType.JSON);
    }

    if (data.action === "getData") {
      var sheet = getOrCreateDatabase();
      var dataRows = sheet.getRange("A:A").getValues();
      var fullStr = "";
      for (var i = 0; i < dataRows.length; i++) {
         if (dataRows[i][0]) fullStr += dataRows[i][0];
      }
      var parsed = fullStr ? JSON.parse(fullStr) : {};
      parsed.quota = currentQuota;
      return ContentService.createTextOutput(JSON.stringify(parsed)).setMimeType(ContentService.MimeType.JSON);
    }

    var to = data.to;
    var subject = data.subject;
    var body = data.body;
    var senderName = data.senderName;
    var fromAlias = data.fromAlias;
    var scheduledTime = data.scheduledTime;
    var attachments = data.attachments || [];

    if (scheduledTime && scheduledTime > new Date().getTime()) {
       var trigger = ScriptApp.newTrigger("sendScheduled").timeBased().at(new Date(scheduledTime)).create();
       var props = PropertiesService.getScriptProperties();
       props.setProperty(trigger.getUniqueId(), JSON.stringify(data));

function getOrCreateDatabase() {
  var props = PropertiesService.getUserProperties();
  var dbId = props.getProperty("MailerSend_DB_ID");
  var ss;
  if (dbId) {
    try { ss = SpreadsheetApp.openById(dbId); } catch(e) { ss = null; }
  }
  if (!ss) {
    ss = SpreadsheetApp.create("MailerSend_Database");
    props.setProperty("MailerSend_DB_ID", ss.getId());
  }
  var sheet = ss.getSheetByName("DataStore");
  if (!sheet) {
    sheet = ss.insertSheet("DataStore");
  }
  return sheet;
}

function sendMailHelper(data) {
   var blobs = [];
   if (data.attachments) {
     data.attachments.forEach(function(att) {
       blobs.push(Utilities.newBlob(Utilities.base64Decode(att.data), att.mimeType, att.name));
     });
   }
   var options = { htmlBody: data.body };
   if (data.senderName) options.name = data.senderName;
   if (data.fromAlias) options.from = data.fromAlias;
   if (blobs.length > 0) options.attachments = blobs;
   
   GmailApp.sendEmail(data.to, data.subject, "", options);
}

function sendScheduled(e) {
   var props = PropertiesService.getScriptProperties();
   var dataStr = props.getProperty(e.triggerUid);
   if (dataStr) {
      sendMailHelper(JSON.parse(dataStr));
      props.deleteProperty(e.triggerUid);
   }
   ScriptApp.getProjectTriggers().forEach(function(t) {
     if (t.getUniqueId() === e.triggerUid) ScriptApp.deleteTrigger(t);
   });
}`;

    const AnimatedCounter = ({ value, label, icon: Icon, colorClass, iconPadding = 'p-3', iconClass = "w-5 h-5 xl:w-[22px] xl:h-[22px]", title }) => {
      const [display, setDisplay] = useState(0);

        let start = display;
        const end = Number(value) || 0;
        const duration = 1000;
        const startT = performance.now();
        const step = (now) => {
          const progress = Math.min((now - startT) / duration, 1);
          setDisplay(Math.floor(start + (end - start) * progress));
          if (progress < 1) requestAnimationFrame(step);
          else setDisplay(end);
        };
        requestAnimationFrame(step);

    const RichTextEditor = ({ value, onChange, t }) => {
      const editorRef = useRef(null);
      const [isHtmlMode, setIsHtmlMode] = useState(false);
      const [modalConfig, setModalConfig] = useState(null);
      const [savedRange, setSavedRange] = useState(null);

const safeGetLocal = (key, fallback) => {
  try {
    const item = localStorage.getItem(key);
    if (!item) return fallback;
    const parsed = JSON.parse(item);
    // Validate data types to prevent corruption from breaking the app
    if (key === 'mailerSend_templates' && !Array.isArray(parsed)) return fallback;
    if (key === 'mailerSend_contacts' && !Array.isArray(parsed)) return fallback;
    if (key === 'mailerSend_logs' && !Array.isArray(parsed)) return fallback;
    if (key === 'mailerSend_stats' && typeof parsed !== 'object') return fallback;
    return parsed;
  } catch (e) {
    console.error(`Error parsing ${key} from local storage:`, e);
    return fallback; // Return default data if corrupted
  }
};

const handleFileUpload = (e) => {
  const files = Array.from(e.target.files);
  let totalSize = attachments.reduce((acc, file) => acc + file.size, 0);
  const newAttachments = [];
  
  files.forEach(file => {
    if (totalSize + file.size > 10 * 1024 * 1024) {
      showNotif(`Skipped ${file.name}: Exceeds 10MB.`, 'error');
      return;
    }
    totalSize += file.size;
    
    const reader = new FileReader();
    reader.onload = (event) => {
      newAttachments.push({
        id: Math.random().toString(36).substr(2, 9),
        name: file.name,
        mimeType: file.type,
        data: event.target.result.split(',')[1],
        size: file.size
      });
      if (newAttachments.length === files.length) {
         setAttachments(prev => [...prev, ...newAttachments]);
      }
    };
    reader.readAsDataURL(file);
  });
};

const sendEmail = async () => {
  if (!gasUrl) return setShowSetup(true);
  if (!to || !subject) return showNotif(t('notif_req'), 'error');

  setIsLoading(true);

  let scheduledTimestamp = null;
  if (scheduleDate && scheduleTime) {
    scheduledTimestamp = new Date(`${scheduleDate}T${scheduleTime}`).getTime();
    if (scheduledTimestamp <= new Date().getTime()) {
      setIsLoading(false);
      return showNotif('Scheduled time must be in the future.', 'error');
    }
  }

  let finalBody = body;
  if (attachSignature && signature) {
    finalBody += `<br/><br/>${signature}`;
  }

  const payload = {
    to, subject, body: finalBody, senderName, fromAlias,
    attachments: attachments.map(({name, mimeType, data}) => ({name, mimeType, data})),
    scheduledTime: scheduledTimestamp
  };

  try {
    const response = await fetch(gasUrl, {
      method: 'POST',
      body: JSON.stringify(payload),
      headers: { 'Content-Type': 'text/plain;charset=utf-8' }
    });

    // Google Script sometimes returns HTML error pages if deployed wrong
    const contentType = response.headers.get("content-type");
    if (contentType && !contentType.includes("application/json")) {
       throw new Error("Invalid response format. Re-deploy script as 'Web App' with access 'Anyone'.");
    }

    const result = await response.json();
    if (result.error) throw new Error(result.error);

    if (result.status === 'scheduled') {
      showNotif(`Email scheduled for ${scheduleDate} ${scheduleTime}`);
      addLog('Email', `Scheduled email to: ${to}`, 'success');
    } else {
      showNotif(t('notif_sent'));
      addLog('Email', `Sent email to: ${to}`, 'success');

      const newSentCount = (stats.sent || 0) + 1;
      const newStats = { ...stats, sent: newSentCount };
      setStats(newStats);
      safeSetLocal('mailerSend_stats', newStats); // Pass object directly

      if (newSentCount === 10) {
        setTimeout(() => confetti({ particleCount: 200, spread: 100, origin: { y: 0.5 }, colors: [COLORS.primary, COLORS.highlight, '#ffffff', '#c0c0c0'] }), 500);
      } else if (newSentCount === 50) {
        setTimeout(() => confetti({ particleCount: 300, spread: 120, origin: { y: 0.5 }, colors: [COLORS.primary, COLORS.highlight, '#ffffff', '#FFD700'] }), 500);
      } else if (newSentCount === 100) {
        setShowGoldPopup(true);
        const duration = 5 * 1000;
        const animationEnd = Date.now() + duration;
        const defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 100 };
        const interval = setInterval(function() {
          const timeLeft = animationEnd - Date.now();
          if (timeLeft <= 0) return clearInterval(interval);
          const particleCount = 50 * (timeLeft / duration);
          confetti(Object.assign({}, defaults, { particleCount, origin: { x: Math.random(), y: Math.random() - 0.2 } }));
        }, 250);
      } else if (newSentCount === 1) {
        setTimeout(() => confetti({ particleCount: 150, spread: 70, origin: { y: 0.5 }, colors: [COLORS.primary, COLORS.highlight, '#ffffff'] }), 500);
      }
    }
    
    if (result.quota !== undefined) {
       setQuota(result.quota);
    }

    setTo('');
    setAttachments([]);
    setScheduleDate('');
    setScheduleTime('');
    setHasConnectionError(false);
  } catch (err) {
    console.error(err);
    const diagMsg = err.message.includes('Unexpected token') || err.message.includes('format') 
        ? 'Backend returned invalid format. Check deployment settings (Anyone).' 
        : `Failed to send: ${err.message}`;
    showNotif(diagMsg, 'error');
    addLog('Error', `Send failed: ${err.message}`, 'error');
    setHasConnectionError(true);
  } finally {
    setIsLoading(false);
  }
};
